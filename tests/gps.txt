
#include <Arduino.h>
#include <SparkFun_u-blox_GNSS_Arduino_Library.h>

SFE_UBLOX_GNSS myGNSS;

HardwareSerial GPS(1);

// ESP32 UART pins (можешь поменять)
#define GPS_RX 16
#define GPS_TX 17

void setup()
{
  Serial.begin(115200);
  delay(1000);

  Serial.println("Starting GNSS...");

  // UART1
  GPS.begin(38400, SERIAL_8N1, GPS_RX, GPS_TX);

  // подключаем библиотеку к UART
  if (!myGNSS.begin(GPS))
  {
    Serial.println("GNSS not detected. Check wiring / baud rate.");
    while (1);
  }

  Serial.println("GNSS connected.");

  // можно ускорить обновления (не обязательно)
  myGNSS.setI2COutput(COM_TYPE_UBX); // не влияет на UART, но иногда помогает конфигу
}

void loop()
{
  if (myGNSS.getPVT() == true)
  {
    long latitude  = myGNSS.getLatitude();
    long longitude = myGNSS.getLongitude();
    long altitude  = myGNSS.getAltitude();

	Serial.println(myGNSS.getSIV());

    Serial.print("Lat: ");
    Serial.print(latitude / 10000000.0);

    Serial.print(" | Lon: ");
    Serial.print(longitude / 10000000.0);

    Serial.print(" | Alt: ");
    Serial.print(altitude / 1000.0);

    Serial.println();
  }
  else
  {
    Serial.println("Waiting for GNSS fix...");
  }

  delay(1000);
}
